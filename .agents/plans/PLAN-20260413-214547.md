```go
// --- Effort ---

type Effort string

const (
    EffortUnspecified Effort = ""
    EffortLow        Effort = "low"
    EffortMedium     Effort = "medium"
    EffortHigh       Effort = "high"
    EffortMax        Effort = "max"
)

func (e Effort) IsEmpty() bool { return e == EffortUnspecified }

func (e Effort) Valid() bool {
    switch e {
    case EffortUnspecified, EffortLow, EffortMedium, EffortHigh, EffortMax:
        return true
    default:
        return false
    }
}

// ToBudget maps this effort to a token budget in [low, high].
// Used by providers that need budget_tokens (Anthropic < 4.6, Bedrock).
// EffortMax maps to the same budget as EffortHigh.
// Returns (0, false) for EffortUnspecified.
func (e Effort) ToBudget(low, high int) (int, bool) {
    switch e {
    case EffortLow:             return low, true
    case EffortMedium:          return low + (high-low)/2, true
    case EffortHigh, EffortMax: return high, true
    default:                    return 0, false
    }
}

// --- ThinkingMode ---

type ThinkingMode string

const (
    ThinkingAuto ThinkingMode = ""
    ThinkingOn   ThinkingMode = "on"
    ThinkingOff  ThinkingMode = "off"
)

func (m ThinkingMode) IsOff() bool { return m == ThinkingOff }
func (m ThinkingMode) IsOn() bool  { return m == ThinkingOn }

func (m ThinkingMode) Valid() bool {
    switch m {
    case ThinkingAuto, ThinkingOn, ThinkingOff:
        return true
    default:
        return false
    }
}
```

```go
// mapEffortAndThinking resolves Effort + ThinkingMode into the provider-specific
// reasoning_effort string. Returns "" when reasoning_effort should be omitted.
func mapEffortAndThinking(model string, effort llm.Effort, thinking llm.ThinkingMode) (string, error) {
    info, err := getModelInfo(model)
    if err != nil {
        return "", err
    }

    // Non-reasoning models: reasoning_effort is always omitted
    if info.Category == categoryNonReasoning {
        return "", nil
    }

    // ThinkingOff: map to "none" where supported, omit elsewhere
    if thinking.IsOff() {
        switch info.Category {
        case categoryGPT51, categoryCodex:
            return "none", nil
        default:
            return "", nil // pre-GPT-5.1, Pro: can't disable, ignore gracefully
        }
    }

    // EffortMax: downgrade per model category
    if effort == llm.EffortMax {
        if info.Category == categoryCodex {
            return "xhigh", nil
        }
        effort = llm.EffortHigh // silent downgrade for all others
    }

    // ThinkingOn + EffortUnspecified: default to high
    if thinking.IsOn() && effort.IsEmpty() {
        return "high", nil
    }

    // EffortUnspecified + ThinkingAuto: omit (let API decide)
    if effort.IsEmpty() {
        return "", nil
    }

    // Direct mapping: low/medium/high are valid OpenAI values
    return string(effort), nil
}
```

Note: removes all the per-category `ThinkingEffortMinimal`/`ThinkingEffortXHigh` translation — those values no longer exist.
